const { analyzeProject } = require("./analyzerEngine");
const { runFix } = require("./autoFixEngine");

async function runOneClickPipeline(project, options = {}) {
  const steps = [];
  const updatedProject = { ...project };

  steps.push({
    key: "upload",
    status: "complete",
    message: "Project received by pipeline."
  });

  let analysis = analyzeProject(updatedProject);

  steps.push({
    key: "analyze",
    status: "complete",
    message: `Analyzer finished with score ${analysis.score} (${analysis.readiness}).`,
    data: analysis
  });

  if (options.autoFix !== false && analysis.issues.length) {
    const appliedFixes = [];

    for (const issue of analysis.issues) {
      if (issue.fix) {
        const result = runFix(issue.fix, updatedProject);
        appliedFixes.push({
          fix: issue.fix,
          result
        });
      }
    }

    analysis = analyzeProject(updatedProject);

    steps.push({
      key: "autofix",
      status: "complete",
      message: `Applied ${appliedFixes.length} automatic fixes.`,
      data: {
        appliedFixes,
        postFixAnalysis: analysis
      }
    });
  } else {
    steps.push({
      key: "autofix",
      status: "skipped",
      message: "No automatic fixes were needed."
    });
  }

  const blockers = analysis.issues.filter((issue) => issue.type === "blocker");

  if (blockers.length) {
    steps.push({
      key: "submission_ready",
      status: "blocked",
      message: "Pipeline stopped because submission blockers remain.",
      data: {
        blockers
      }
    });

    return {
      ok: false,
      stage: "blocked",
      project: updatedProject,
      analysis,
      steps
    };
  }

  steps.push({
    key: "submission_ready",
    status: "complete",
    message: "Project is ready for billing and submission checks."
  });

  return {
    ok: true,
    stage: "ready_for_submission",
    project: updatedProject,
    analysis,
    steps
  };
}

module.exports = { runOneClickPipeline };
