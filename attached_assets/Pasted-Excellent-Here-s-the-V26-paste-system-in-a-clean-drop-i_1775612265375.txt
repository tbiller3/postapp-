src/utils/sourceSync.ts

export type DetectedSubmissionData = {
  appName?: string;
  subtitle?: string;
  bundleId?: string;
  version?: string;
  buildNumber?: string;
  category?: string;
  ageRating?: string;
  keywords?: string;
  supportUrl?: string;
  privacyPolicyUrl?: string;
  description?: string;
};

export function normalizeDetectedData(raw: Partial<DetectedSubmissionData>): DetectedSubmissionData {
  return {
    appName: raw.appName?.trim() || "",
    subtitle: raw.subtitle?.trim() || "",
    bundleId: raw.bundleId?.trim() || "",
    version: raw.version?.trim() || "",
    buildNumber: raw.buildNumber?.trim() || "",
    category: raw.category?.trim() || "",
    ageRating: raw.ageRating?.trim() || "",
    keywords: raw.keywords?.trim() || "",
    supportUrl: raw.supportUrl?.trim() || "",
    privacyPolicyUrl: raw.privacyPolicyUrl?.trim() || "",
    description: raw.description?.trim() || ""
  };
}

export function getFieldStatus(currentValue?: string, detectedValue?: string) {
  const current = (currentValue || "").trim();
  const detected = (detectedValue || "").trim();

  if (!current && !detected) return "missing";
  if (!current && detected) return "missing";
  if (current && !detected) return "manual";
  if (current === detected) return "verified";
  return "modified";
}

src/state/submissionStore.ts

import { create } from "zustand";
import { normalizeDetectedData, type DetectedSubmissionData } from "../utils/sourceSync";

type PricingModel = "free" | "paid" | "freemium" | "subscription";

type SubmissionFields = {
  appName: string;
  subtitle: string;
  bundleId: string;
  version: string;
  buildNumber: string;
  category: string;
  ageRating: string;
  keywords: string;
  supportUrl: string;
  privacyPolicyUrl: string;
  description: string;
};

type PricingState = {
  model: PricingModel;
  hasIAP: boolean;
  hasSubscriptions: boolean;
  freeTrial: boolean;
  priceTier: string;
  notes: string;
};

type SubmissionStore = {
  plan: "free" | "pro" | "studio";
  fields: SubmissionFields;
  detected: DetectedSubmissionData;
  pricing: PricingState;
  setField: (key: keyof SubmissionFields, value: string) => void;
  setPricingField: (key: keyof PricingState, value: string | boolean) => void;
  syncDetected: (payload: Partial<DetectedSubmissionData>) => void;
  applyDetectedValue: (key: keyof SubmissionFields) => void;
  applyAllDetectedValues: () => void;
  loadDemoSubmission: () => void;
  getCompletionStats: () => { complete: number; missing: number; total: number; percent: number };
};

const initialFields: SubmissionFields = {
  appName: "",
  subtitle: "",
  bundleId: "",
  version: "",
  buildNumber: "",
  category: "",
  ageRating: "",
  keywords: "",
  supportUrl: "",
  privacyPolicyUrl: "",
  description: ""
};

const initialPricing: PricingState = {
  model: "free",
  hasIAP: false,
  hasSubscriptions: false,
  freeTrial: false,
  priceTier: "",
  notes: ""
};

export const useSubmissionStore = create<SubmissionStore>((set, get) => ({
  plan: "pro",

  fields: initialFields,
  detected: {},
  pricing: initialPricing,

  setField: (key, value) =>
    set((state) => ({
      fields: {
        ...state.fields,
        [key]: value
      }
    })),

  setPricingField: (key, value) =>
    set((state) => ({
      pricing: {
        ...state.pricing,
        [key]: value
      }
    })),

  syncDetected: (payload) =>
    set(() => ({
      detected: normalizeDetectedData(payload)
    })),

  applyDetectedValue: (key) =>
    set((state) => {
      const detectedValue = state.detected[key];
      if (typeof detectedValue !== "string") return state;

      return {
        fields: {
          ...state.fields,
          [key]: detectedValue
        }
      };
    }),

  applyAllDetectedValues: () =>
    set((state) => {
      const merged = { ...state.fields };

      (Object.keys(state.fields) as Array<keyof SubmissionFields>).forEach((key) => {
        const detectedValue = state.detected[key];
        if (typeof detectedValue === "string" && detectedValue.trim()) {
          merged[key] = detectedValue;
        }
      });

      return { fields: merged };
    }),

  loadDemoSubmission: () =>
    set(() => ({
      fields: {
        appName: "WAIT Recovery Companion",
        subtitle: "Pause through urges",
        bundleId: "com.wait.recovery",
        version: "1.0.0",
        buildNumber: "1",
        category: "Health & Fitness",
        ageRating: "17+",
        keywords: "recovery,addiction,sobriety,calm,urge support,mental health",
        supportUrl: "https://wait.example.com/support",
        privacyPolicyUrl: "https://wait.example.com/privacy",
        description:
          "WAIT Recovery Companion helps users pause through cravings and urges with guided support, calming tools, and progress tracking."
      },
      detected: {
        appName: "WAIT Recovery Companion",
        version: "1.0.0"
      },
      pricing: {
        model: "free",
        hasIAP: false,
        hasSubscriptions: false,
        freeTrial: false,
        priceTier: "",
        notes: ""
      }
    })),

  getCompletionStats: () => {
    const { fields } = get();
    const keys = Object.keys(fields) as Array<keyof SubmissionFields>;
    const complete = keys.filter((key) => String(fields[key]).trim()).length;
    const total = keys.length;
    const missing = total - complete;
    const percent = total ? Math.round((complete / total) * 100) : 0;

    return { complete, missing, total, percent };
  }
}));

3. Add this component

src/components/SubmissionFieldCard.tsx

import { useSubmissionStore } from "../state/submissionStore";
import { getFieldStatus } from "../utils/sourceSync";

type Props = {
  label: string;
  fieldKey:
    | "appName"
    | "subtitle"
    | "bundleId"
    | "version"
    | "buildNumber"
    | "category"
    | "ageRating"
    | "keywords"
    | "supportUrl"
    | "privacyPolicyUrl"
    | "description";
  placeholder?: string;
  multiline?: boolean;
};

export default function SubmissionFieldCard({
  label,
  fieldKey,
  placeholder,
  multiline = false
}: Props) {
  const { fields, detected, setField, applyDetectedValue } = useSubmissionStore();

  const currentValue = fields[fieldKey] || "";
  const detectedValue = detected[fieldKey] || "";
  const status = getFieldStatus(currentValue, detectedValue);

  return (
    <div className="v26-card">
      <div className="v26-card-top">
        <div>
          <div className="v26-card-label">{label}</div>
          <div className="v26-card-detected">
            {detectedValue ? (
              <>
                Detected: <span>{detectedValue}</span>
              </>
            ) : (
              <em>No detected value</em>
            )}
          </div>
        </div>

        <div className={`v26-chip ${status}`}>{status}</div>
      </div>

      {multiline ? (
        <textarea
          className="v26-input v26-textarea"
          value={currentValue}
          placeholder={placeholder}
          onChange={(e) => setField(fieldKey, e.target.value)}
        />
      ) : (
        <input
          className="v26-input"
          value={currentValue}
          placeholder={placeholder}
          onChange={(e) => setField(fieldKey, e.target.value)}
        />
      )}

      <div className="v26-actions">
        {!!detectedValue && (
          <button className="v26-btn-secondary" onClick={() => applyDetectedValue(fieldKey)}>
            Use detected value
          </button>
        )}
      </div>
    </div>
  );
}

src/components/PricingPanel.tsx

import { useSubmissionStore } from "../state/submissionStore";

export default function PricingPanel() {
  const { pricing, setPricingField, plan } = useSubmissionStore();

  const autoPricingEnabled = plan !== "free";

  return (
    <div className="v26-panel">
      <div className="v26-panel-head">
        <h3>Pricing & Monetization</h3>
        <div className="v26-plan-pill">{plan.toUpperCase()}</div>
      </div>

      <div className="v26-form-group">
        <label>Pricing model</label>
        <select
          className="v26-input"
          value={pricing.model}
          onChange={(e) => setPricingField("model", e.target.value)}
        >
          <option value="free">Free</option>
          <option value="paid">Paid</option>
          <option value="freemium">Freemium</option>
          <option value="subscription">Subscription</option>
        </select>
      </div>

      {(pricing.model === "paid" || pricing.model === "subscription") && (
        <div className="v26-form-group">
          <label>Price tier</label>
          <input
            className="v26-input"
            placeholder="Example: Tier 1"
            value={pricing.priceTier}
            onChange={(e) => setPricingField("priceTier", e.target.value)}
          />
        </div>
      )}

      <div className="v26-checkbox-list">
        <label className="v26-check">
          <input
            type="checkbox"
            checked={pricing.hasIAP}
            onChange={(e) => setPricingField("hasIAP", e.target.checked)}
          />
          <span>Includes in-app purchases</span>
        </label>

        <label className="v26-check">
          <input
            type="checkbox"
            checked={pricing.hasSubscriptions}
            onChange={(e) => setPricingField("hasSubscriptions", e.target.checked)}
          />
          <span>Uses subscriptions</span>
        </label>

        <label className="v26-check">
          <input
            type="checkbox"
            checked={pricing.freeTrial}
            onChange={(e) => setPricingField("freeTrial", e.target.checked)}
          />
          <span>Free trial offered</span>
        </label>
      </div>

      <div className="v26-form-group">
        <label>Pricing notes</label>
        <textarea
          className="v26-input v26-textarea"
          placeholder="Terms, restore purchases, disclosure notes..."
          value={pricing.notes}
          onChange={(e) => setPricingField("notes", e.target.value)}
        />
      </div>

      {!autoPricingEnabled && (
        <div className="v26-note">
          Upgrade to Pro to enable advanced pricing automation and listing sync.
        </div>
      )}
    </div>
  );
}
