// /src/state/submissionStore.ts

import { create } from "zustand";

export const useSubmissionStore = create((set) => ({
  data: {
    appName: "",
    bundleId: "",
    version: "",
    buildNumber: "",
    description: "",
    keywords: "",
    privacyPolicyUrl: "",
    supportUrl: "",
    pricing: {
      model: "free",
      priceTier: "",
      hasIAP: false,
      subscription: false,
      terms: ""
    }
  },

  setField: (key, value) =>
    set((state) => ({
      data: { ...state.data, [key]: value }
    })),

  setPricing: (pricing) =>
    set((state) => ({
      data: { ...state.data, pricing }
    }))
}));

⚙️ 3. VERIFICATION ENGINE

// /src/utils/verificationEngine.ts

export function verifyField({ current, detected }) {
  if (!current) return "missing";
  if (!detected) return "needs-review";
  if (current === detected) return "complete";
  return "mismatch";
}

🧱 4. FIELD ROW (UI)

// /src/components/FieldRow.tsx

import { useSubmissionStore } from "../state/submissionStore";
import { verifyField } from "../utils/verificationEngine";

export default function FieldRow({ label, fieldKey, detected }) {
  const { data, setField } = useSubmissionStore();

  const value = data[fieldKey] || "";
  const status = verifyField({ current: value, detected });

  return (
    <div className="field-row">
      <div className="field-label">{label}</div>

      <input
        value={value}
        onChange={(e) => setField(fieldKey, e.target.value)}
      />

      <div className={`status ${status}`}>{status}</div>

      {detected && (
        <button onClick={() => setField(fieldKey, detected)}>
          Use detected
        </button>
      )}
    </div>
  );
}

💰 6. PRICING EDITOR

// /src/components/PricingEditor.tsx

import { useSubmissionStore } from "../state/submissionStore";

export default function PricingEditor() {
  const { data, setPricing } = useSubmissionStore();

  return (
    <div>
      <h3>Pricing</h3>

      <select
        value={data.pricing.model}
        onChange={(e) =>
          setPricing({ ...data.pricing, model: e.target.value })
        }
      >
        <option value="free">Free</option>
        <option value="paid">Paid</option>
        <option value="subscription">Subscription</option>
      </select>

      {data.pricing.model === "paid" && (
        <input
          placeholder="Price Tier"
          value={data.pricing.priceTier}
          onChange={(e) =>
            setPricing({ ...data.pricing, priceTier: e.target.value })
          }
        />
      )}
    </div>
  );
}
